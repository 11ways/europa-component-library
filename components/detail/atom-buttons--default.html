@mixin _white_space_reset_mixin() {

  padding-bottom: 0.5em;

  white-space: normal;
}

.btn {
  @extend .hidden-print;

  border-radius: 0;
  border-width: 2px;
  font-size: $font-size-little;

    line-height: inherit;

  &.btn-secondary {
