{# Define macro to make it easily accessible from elsewhere #}
{% macro breadcrumb(segments=[], modifier="") %}
<nav class="breadcrumb{{ " breadcrumb--" + modifier if modifier|length}}" role="navigation" aria-label="breadcrumbs">
  <span class="sr-only">You are here:</span>
  <ol class="breadcrumb__segments-wrapper">
    {% for segment in segments %}
    <li class="breadcrumb__segment{{ " breadcrumb__segment--first" if loop.first }}{{ " breadcrumb__segment--last" if loop.last }}">
      <a href="{{ segment.target }}" class="breadcrumb__link{{ " is-internal" if segment.isInternal }}">
        <span class="breadcrumb__text">{{ segment.title }}</span>
      </a>
    </li>
    {% endfor %}
  </ol>
  <span class="breadcrumb__btn-separator" style="display: none;">...</span>
</nav>
{% endmacro %}
